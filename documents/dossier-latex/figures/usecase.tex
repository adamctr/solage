\begin{tikzpicture}[

  lnk/.style={solageGray, thin},
  gen/.style={-{Triangle[open,length=3mm,width=3mm]}, solageDark, semithick},
  actorlbl/.style={font=\scriptsize\bfseries, align=center},
  actor/.pic={

    \draw[thick] (0,-3mm) -- (0,-1.2);
    \draw[thick] (-5mm,-0.75) -- (5mm,-0.75);
    \draw[thick] (0,-1.2) -- (-5mm,-1.8);
    \draw[thick] (0,-1.2) -- (5mm,-1.8);
  }

% --- Frontière du système ---
\node[draw=solageDark, rounded corners, minimum width=7cm, minimum height=12.4cm,
      label={[font=\bfseries]north:Solage}] (sys) at (7.3,-3.2) {};

% --- Cas d'utilisation ---
\node[uc] (inscr)  at (7.3,2.1)  {S'inscrire};
\node[uc] (conn)   at (7.3,1.1)  {Se connecter};

\draw[lnk] (0,1.3) -- (conn.west);

\draw[lnk] (0,-7.9) -- (moder.west);
